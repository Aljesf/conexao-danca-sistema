function formatarReaisDeCentavos(
  valorCentavos: number | null | undefined
): string {
  if (valorCentavos == null || Number.isNaN(valorCentavos)) return "Ś";
  const valor = valorCentavos / 100;
  return valor.toLocaleString("pt-BR", {
    style: "currency",
    currency: "BRL",
    minimumFractionDigits: 2,
  });
}

function formatarData(dateStr: string | null | undefined): string {
  if (!dateStr) return "Ś";
  const d = new Date(dateStr);
  if (Number.isNaN(d.getTime())) return "Ś";
  return d.toLocaleString("pt-BR");
}
