✅ PASSO 2 — Ajustar a tabela turma_aluno

No MESMO arquivo de migration (`20251202_etapa1_create_matriculas_and_adjust_turma_aluno.sql`), após o bloco de criação da tabela `matriculas`, adicione o bloco abaixo para ajustar `turma_aluno`:

```sql
-- 6) AJUSTES NA TABELA TURMA_ALUNO

-- 6.1) GARANTIR FK EXPLÍCITA DE aluno_pessoa_id -> pessoas.id
DO $$
BEGIN
IF NOT EXISTS (
  SELECT 1 FROM pg_constraint WHERE conname = 'turma_aluno_aluno_pessoa_id_fkey'
) THEN
  ALTER TABLE public.turma_aluno
  ADD CONSTRAINT turma_aluno_aluno_pessoa_id_fkey
  FOREIGN KEY (aluno_pessoa_id) REFERENCES public.pessoas(id)
  ON UPDATE CASCADE ON DELETE RESTRICT;
END IF;
END$$;

-- 6.2) ADICIONAR COLUNA matricula_id (SE AINDA NÃO EXISTIR)
DO $$
BEGIN
IF NOT EXISTS (
  SELECT 1 FROM information_schema.columns
  WHERE table_schema = 'public'
    AND table_name = 'turma_aluno'
    AND column_name = 'matricula_id'
) THEN
  ALTER TABLE public.turma_aluno
  ADD COLUMN matricula_id BIGINT NULL;
END IF;
END$$;

-- 6.3) FK matricula_id -> matriculas(id)
DO $$
BEGIN
IF NOT EXISTS (
  SELECT 1 FROM pg_constraint WHERE conname = 'turma_aluno_matricula_id_fkey'
) THEN
  ALTER TABLE public.turma_aluno
  ADD CONSTRAINT turma_aluno_matricula_id_fkey
  FOREIGN KEY (matricula_id) REFERENCES public.matriculas(id)
  ON UPDATE CASCADE ON DELETE SET NULL;
END IF;
END$$;

-- 6.4) ÍNDICES DE APOIO
CREATE INDEX IF NOT EXISTS idx_turma_aluno_matricula
ON public.turma_aluno (matricula_id);

CREATE INDEX IF NOT EXISTS idx_turma_aluno_pessoa
ON public.turma_aluno (aluno_pessoa_id);
```
